# 🎲 vue组件销毁时去除定时器

mounted(){
  const timer = setInterval(() =>{
    console.log("lalala")
  }, 500)

  this.$once('hook:beforeDestroy', () => clearInterval(timer))
}

# 🎲 in 用法

集合遍历的效率为：hash > for(;😉 > for(in)

# 1. 判断属性属于对象

var map = {
  a: 1,
  b: 2,
}

if('a' in map){
  return true
} else {
  return false
}

# 2. for in 遍历对象属性

for(var p in map){
  console.log(p + ':' + map[p])
}

# 3. in 可以用来判断

if(k == 'a' || k == 'b' || k == 'c') {}
// 可以写成

if( k in {'a':'', 'b':'', 'c':''})

  window.location.href="about:blank";  
  window.close();  
} else {
  window.opener = null;  
  window.open("", "_self");  
  window.close();  
}

# 🎲 取余

// 丢弃小数部分,保留整数部分
parseInt(7/2)　　// 3

// 向上取整,有小数就整数部分加1
Math.ceil(7/2)　　// 4

// 向下取整,丢弃小数部分
Math.floor(7/2)　　// 3

// 四舍五入
Math.round(7/2)　　// 3

// 1. 取余
7%2　　// 1
